import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np
from scipy.stats import norm

# Load the dataset
df = pd.read_csv("./IQAnalysis.csv")
df.head()


# Create subplots for each bell curve
fig, axes = plt.subplots(1, 3, figsize=(15, 5))

# Bell Curve for Brain Circumference
mu_brain, std_brain = np.mean(df['Brain']), np.std(df['Brain'])
xmin_brain, xmax_brain = min(df['Brain']), max(df['Brain'])
x_brain = np.linspace(xmin_brain, xmax_brain, 100)
p_brain = norm.pdf(x_brain, mu_brain, std_brain)

axes[0].hist(df['Brain'], bins=10, density=True, color='blue', alpha=0.7, label='Brain Circumference')
axes[0].plot(x_brain, p_brain, 'k', linewidth=2, label='Fit Normal Curve')
axes[0].set_title('Bell Curve for Brain Circumference')
axes[0].set_xlabel('Brain Circumference (cm)')
axes[0].set_ylabel('Probability Density')
axes[0].legend()

# Bell Curve for Height
mu_height, std_height = np.mean(df['Height']), np.std(df['Height'])
xmin_height, xmax_height = min(df['Height']), max(df['Height'])
x_height = np.linspace(xmin_height, xmax_height, 100)
p_height = norm.pdf(x_height, mu_height, std_height)

axes[1].hist(df['Height'], bins=10, density=True, color='green', alpha=0.7, label='Height')
axes[1].plot(x_height, p_height, 'k', linewidth=2, label='Fit Normal Curve')
axes[1].set_title('Bell Curve for Height')
axes[1].set_xlabel('Height (inches)')
axes[1].set_ylabel('Probability Density')
axes[1].legend()

# Bell Curve for Weight
mu_weight, std_weight = np.mean(df['Weight']), np.std(df['Weight'])
xmin_weight, xmax_weight = min(df['Weight']), max(df['Weight'])
x_weight = np.linspace(xmin_weight, xmax_weight, 100)
p_weight = norm.pdf(x_weight, mu_weight, std_weight)

axes[2].hist(df['Weight'], bins=10, density=True, color='red', alpha=0.7, label='Weight')
axes[2].plot(x_weight, p_weight, 'k', linewidth=2, label='Fit Normal Curve')
axes[2].set_title('Bell Curve for Weight')
axes[2].set_xlabel('Weight (lbs)')
axes[2].set_ylabel('Probability Density')
axes[2].legend()

plt.tight_layout()
plt.show()
